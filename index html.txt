    <script>
        document.addEventListener('DOMContentLoaded', () => {
            
            // 1. Sticky Navigation Effect
            const navbar = document.getElementById('navbar');
            window.addEventListener('scroll', () => {
                if (window.scrollY > 50) {
                    navbar.classList.add('scrolled');
                } else {
                    navbar.classList.remove('scrolled');
                }
            });

            // 2. Mobile Menu Toggle
            const mobileBtn = document.getElementById('mobile-menu-btn');
            const mobileMenu = document.getElementById('mobile-menu');
            
            mobileBtn.addEventListener('click', () => {
                mobileMenu.classList.toggle('hidden');
            });

            // Close mobile menu when clicking a link
            mobileMenu.querySelectorAll('a').forEach(link => {
                link.addEventListener('click', () => {
                    mobileMenu.classList.add('hidden');
                });
            });

            // 3. Form Validation
            const form = document.getElementById('cakeOrderForm');
            
            form.addEventListener('submit', (e) => {
                e.preventDefault();
                let isValid = true;

                // Name Validation
                const name = document.getElementById('name');
                const nameError = document.getElementById('nameError');
                if (name.value.trim() === '') {
                    nameError.classList.remove('hidden');
                    name.classList.add('border-red-500');
                    isValid = false;
                } else {
                    nameError.classList.add('hidden');
                    name.classList.remove('border-red-500');
                }

                // Phone Validation (Simple check for numbers and length)
                const phone = document.getElementById('phone');
                const phoneError = document.getElementById('phoneError');
                const phoneRegex = /^[\d\s\-\+\(\)]+$/;
                if (!phoneRegex.test(phone.value) || phone.value.length < 10) {
                    phoneError.classList.remove('hidden');
                    phone.classList.add('border-red-500');
                    isValid = false;
                } else {
                    phoneError.classList.add('hidden');
                    phone.classList.remove('border-red-500');
                }

                // Cake Type Validation
                const cakeType = document.getElementById('cakeType');
                const cakeError = document.getElementById('cakeError');
                if (cakeType.value === '') {
                    cakeError.classList.remove('hidden');
                    cakeType.classList.add('border-red-500');
                    isValid = false;
                } else {
                    cakeError.classList.add('hidden');
                    cakeType.classList.remove('border-red-500');
                }

                // If valid, show success message (simulated)
                if (isValid) {
                    alert('Thank you! Your order request has been submitted. We will contact you shortly!');
                    form.reset();
                }
            });

            // Remove error styles on input
            ['name', 'phone', 'cakeType'].forEach(id => {
                document.getElementById(id).addEventListener('input', function() {
                    this.classList.remove('border-red-500');
                    document.getElementById(id + 'Error').classList.add('hidden');
                });
            });
        });
    </script>
